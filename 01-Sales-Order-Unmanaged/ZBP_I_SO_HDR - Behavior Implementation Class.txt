CLASS lcl_buffer DEFINITION.
  PUBLIC SECTION.
    TYPES: BEGIN OF ts_buffer,
             draft_uuid TYPE sysuuid_x16, " Use this as the key
             header     TYPE zi_so_hdr,
             items      TYPE TABLE OF zi_so_item WITH EMPTY KEY,
             partners   TYPE TABLE OF zi_so_partner WITH EMPTY KEY,
           END OF ts_buffer.

    CLASS-DATA: mt_buffer TYPE TABLE OF ts_buffer. " This holds our SO data in memory

    CLASS-METHODS:
      clear,
      get_instance_data
        IMPORTING iv_vbeln       TYPE vbeln_va
        RETURNING VALUE(rs_data) TYPE ts_buffer,
      get_by_uuid
        IMPORTING iv_uuid        TYPE sysuuid_x16
        RETURNING VALUE(rs_data) TYPE ts_buffer.
ENDCLASS.

CLASS lcl_buffer IMPLEMENTATION.
  METHOD clear.
    FREE mt_buffer.
  ENDMETHOD.

  " Helper to find or create a header entry in memory
  METHOD get_instance_data.

    READ TABLE mt_buffer INTO rs_data WITH KEY header-vbeln = iv_vbeln.

  ENDMETHOD.
  METHOD get_by_uuid.

    READ TABLE mt_buffer INTO rs_data WITH KEY draft_uuid = iv_uuid.

  ENDMETHOD.

ENDCLASS.

CLASS lhc_zi_so_hdr IMPLEMENTATION.

  METHOD get_instance_authorizations.
  ENDMETHOD.

  METHOD create.

    LOOP AT entities ASSIGNING FIELD-SYMBOL(<lfs_entity>).
      " 1. Initialize a new buffer entry
      TRY.
         " 1. Generate the UUID
          DATA(lv_new_uuid) = cl_system_uuid=>create_uuid_x16_static( ).

         " 2. Store in Buffer
          INSERT VALUE #( draft_uuid = lv_new_uuid
                          header     = CORRESPONDING #( <lfs_entity> ) )
                          INTO TABLE lcl_buffer=>mt_buffer.

        " 3. THE CRITICAL FIX: Map %cid to %pid
        " Do NOT just map vbeln. RAP needs %pid to 'remember' this record.
        INSERT VALUE #( %cid = <lfs_entity>-%cid
                        %pid = lv_new_uuid ) INTO TABLE mapped-zi_so_hdr.
        CATCH cx_uuid_error.
          "handle exception
      ENDTRY.

    ENDLOOP.

  ENDMETHOD.

  METHOD update.

    LOOP AT entities ASSIGNING FIELD-SYMBOL(<lfs_entity>).
      READ TABLE lcl_buffer=>mt_buffer ASSIGNING FIELD-SYMBOL(<ls_so>)
        WITH KEY header-vbeln = <lfs_entity>-vbeln.
      IF sy-subrc = 0.
        " Update only the fields provided by the UI
        <ls_so>-header = CORRESPONDING #( BASE ( <ls_so>-header ) <lfs_entity> ).
      ENDIF.
    ENDLOOP.

  ENDMETHOD.

  METHOD delete.
  ENDMETHOD.

  METHOD read.

    " This allows the UI to display the data
    IF keys IS NOT INITIAL.
      SELECT * FROM vbak FOR ALL ENTRIES IN @keys
        WHERE vbeln = @keys-vbeln
        INTO TABLE @DATA(lt_vbak).
      result = CORRESPONDING #( lt_vbak MAPPING vbeln = vbeln ).
    ENDIF.

  ENDMETHOD.

  METHOD lock.
  ENDMETHOD.

  METHOD rba_item.

    IF keys_rba IS NOT INITIAL.
      " 1. Select items from the standard table VBAP
      SELECT * FROM vbap
        FOR ALL ENTRIES IN @keys_rba
        WHERE vbeln = @keys_rba-vbeln
        INTO TABLE @DATA(lt_vbap).

      " 2. Map results back to the target entity (Items)
      result = CORRESPONDING #( lt_vbap MAPPING vbeln = vbeln
                                                posnr = posnr
                                                matnr = matnr ).

      " 3. Fill the link table (RAP needs this to know which Item belongs to which Header)
      LOOP AT lt_vbap ASSIGNING FIELD-SYMBOL(<ls_vbap>).
        INSERT VALUE #( source-vbeln = <ls_vbap>-vbeln
                        target-vbeln = <ls_vbap>-vbeln
                        target-posnr = <ls_vbap>-posnr ) INTO TABLE association_links.
      ENDLOOP.
    ENDIF.

  ENDMETHOD.

  METHOD rba_partner.

    SELECT vbeln, parvw FROM vbpa
      FOR ALL ENTRIES IN @keys_rba
      WHERE vbeln = @keys_rba-vbeln
      INTO TABLE @DATA(lt_vbpa).

    result = CORRESPONDING #( lt_vbpa MAPPING vbeln = vbeln
                                              parvw = parvw ).

    LOOP AT lt_vbpa ASSIGNING FIELD-SYMBOL(<ls_vbpa>).
      APPEND VALUE #(
          source-vbeln = <ls_vbpa>-vbeln
          target-vbeln = <ls_vbpa>-vbeln
          target-parvw = <ls_vbpa>-parvw
      ) TO association_links.
    ENDLOOP.

  ENDMETHOD.

  METHOD cba_item.

    DATA lv_max_posnr TYPE posnr_va VALUE '000010'.

    LOOP AT entities_cba ASSIGNING FIELD-SYMBOL(<ls_cba>).
      " Find the Header in our buffer
      READ TABLE lcl_buffer=>mt_buffer ASSIGNING FIELD-SYMBOL(<ls_so>)
        WITH KEY draft_uuid = <ls_cba>-%pid.

      IF sy-subrc = 0.
        LOOP AT <ls_cba>-%target ASSIGNING FIELD-SYMBOL(<ls_item>).

          " Check existing items in the buffer for this header
          IF <ls_so>-items IS INITIAL.
            lv_max_posnr = '000010'.
          ELSE.
            " Find the highest current position number
            SELECT MAX( posnr ) FROM @<ls_so>-items AS itm INTO @lv_max_posnr.
            lv_max_posnr = lv_max_posnr + 10.
          ENDIF.

          DATA(ls_new_item) = CORRESPONDING zi_so_item( <ls_item> ).
          ls_new_item-posnr = |{ lv_max_posnr ALPHA = IN }|. " Ensure leading zeros
          ls_new_item-vbeln = <ls_cba>-vbeln.

          " Add Item to the Header's item table in buffer
          INSERT CORRESPONDING #( ls_new_item ) INTO TABLE <ls_so>-items.

          " Map the result
          INSERT VALUE #( %cid  = <ls_item>-%cid
                          %pid  = ls_new_item-posnr
                          vbeln = ls_new_item-vbeln
                          posnr = ls_new_item-posnr ) INTO TABLE mapped-zi_so_item.
        ENDLOOP.
      ENDIF.
    ENDLOOP.

  ENDMETHOD.

  METHOD cba_partner.

  LOOP AT entities_cba ASSIGNING FIELD-SYMBOL(<ls_cba>).
    " FIX: Use %pid (draft_uuid) instead of vbeln for new records
    READ TABLE lcl_buffer=>mt_buffer ASSIGNING FIELD-SYMBOL(<ls_so>)
      WITH KEY draft_uuid = <ls_cba>-%pid.

    IF sy-subrc = 0.
      LOOP AT <ls_cba>-%target ASSIGNING FIELD-SYMBOL(<ls_partner>).
        INSERT CORRESPONDING #( <ls_partner> ) INTO TABLE <ls_so>-partners.

        INSERT VALUE #( %cid  = <ls_partner>-%cid
                        %pid  = <ls_cba>-%pid
                        vbeln = <ls_partner>-vbeln
                        parvw = <ls_partner>-parvw ) INTO TABLE mapped-zi_so_partner.
      ENDLOOP.
    ENDIF.
  ENDLOOP.

  ENDMETHOD.

  METHOD set_defaults.

    READ ENTITIES OF zi_so_hdr IN LOCAL MODE
       ENTITY zi_so_hdr
       ALL FIELDS WITH CORRESPONDING #( keys )
       RESULT DATA(lt_orders).

    MODIFY ENTITIES OF zi_so_hdr IN LOCAL MODE
      ENTITY zi_so_hdr
      UPDATE FIELDS ( waerk auart vkorg vtweg spart )     "All fields to be defaulted
      WITH VALUE #( FOR order IN lt_orders (
                    %tky  = order-%tky
                    auart = 'ZOR'        "Default Order Type
                    vkorg = 'ZA01'
                    vtweg = 'Z1'
                    spart = 'Z1'
                    waerk = 'ZAR' ) ).   "Default Currency

  ENDMETHOD.

  METHOD finalize_buffer_on_save.

    " FIX: UNCOMMENTED — This is critical! Without this, the buffer is empty at save time.

    " 1. Read the Header Draft data
    READ ENTITIES OF zi_so_hdr IN LOCAL MODE
      ENTITY zi_so_hdr ALL FIELDS WITH CORRESPONDING #( keys )
      RESULT DATA(lt_headers).

    " 2. Read the Item Draft data
    READ ENTITIES OF zi_so_hdr IN LOCAL MODE
      ENTITY zi_so_hdr BY \_item ALL FIELDS WITH CORRESPONDING #( keys )
      RESULT DATA(lt_items).

    " 3. Read the Partner Draft data
    READ ENTITIES OF zi_so_hdr IN LOCAL MODE
      ENTITY zi_so_hdr BY \_partner ALL FIELDS WITH CORRESPONDING #( keys )
      RESULT DATA(lt_partners).

    " 4. Re-populate the buffer so adjust_numbers has data
    LOOP AT lt_headers INTO DATA(ls_hdr).
      DATA(lv_uuid) = ls_hdr-%tky-%pid.

      " Check if entry already exists in buffer (from create/update)
      READ TABLE lcl_buffer=>mt_buffer ASSIGNING FIELD-SYMBOL(<ls_existing>)
        WITH KEY draft_uuid = lv_uuid.

      IF sy-subrc = 0.
        " Update existing buffer entry with latest draft data
        <ls_existing>-header =   CORRESPONDING #( ls_hdr ).
        <ls_existing>-items =    CORRESPONDING #( lt_items ).
        <ls_existing>-partners = CORRESPONDING #( lt_partners ).
      ELSE.
        " Insert new buffer entry
        INSERT VALUE #(
            draft_uuid = lv_uuid
            header     = CORRESPONDING #( ls_hdr )
            items      = CORRESPONDING #( lt_items )
            partners   = CORRESPONDING #( lt_partners )
        ) INTO TABLE lcl_buffer=>mt_buffer.
      ENDIF.
    ENDLOOP.

  ENDMETHOD.

ENDCLASS.

  METHOD calculate_header_total.

    DATA: lt_item_updates TYPE TABLE FOR UPDATE zi_so_hdr\\zi_so_item,
          lv_hdr_tot      TYPE netwr_ak.

    " 1. Read the Items being modified to get their keys and data
    READ ENTITIES OF zi_so_hdr IN LOCAL MODE
      ENTITY zi_so_item
      ALL FIELDS WITH CORRESPONDING #( keys )
      RESULT DATA(lt_items).

    " 2. Get the Parent Header Link (to find which header to update)
    READ ENTITIES OF zi_so_hdr IN LOCAL MODE
      ENTITY zi_so_item BY \_header
      FROM CORRESPONDING #( keys )
      LINK DATA(lt_link).

    " 3. Calculate Item Totals and collect updates in a single table
    LOOP AT lt_items ASSIGNING FIELD-SYMBOL(<ls_item>).
      DATA(lv_item_total) = <ls_item>-kwmeng * <ls_item>-netpr.

      APPEND VALUE #( %tky  = <ls_item>-%tky
                      netwr = lv_item_total ) TO lt_item_updates.
    ENDLOOP.

    " 4. Perform BULK Update for Items (Prevents recursion/dumps)
    IF lt_item_updates IS NOT INITIAL.
      MODIFY ENTITIES OF zi_so_hdr IN LOCAL MODE
        ENTITY zi_so_item
        UPDATE FIELDS ( netwr ) WITH lt_item_updates.
    ENDIF.

    " 5. Recalculate Header Total (Sum ALL items for this specific header)
    IF lt_link IS NOT INITIAL.
      " We take the first header link for simplicity (assuming one header per batch)
      DATA(ls_header_link) = lt_link[ 1 ].

      " Read ALL items belonging to this Header (not just the ones being modified)
      READ ENTITIES OF zi_so_hdr IN LOCAL MODE
        ENTITY zi_so_hdr BY \_item
        FIELDS ( netwr ) WITH VALUE #( ( %tky = ls_header_link-target ) )
        RESULT DATA(lt_all_items).

      LOOP AT lt_all_items INTO DATA(ls_row).
        lv_hdr_tot += ls_row-netwr.
      ENDLOOP.

      " 6. Update the Header with the total sum
      MODIFY ENTITIES OF zi_so_hdr IN LOCAL MODE
        ENTITY zi_so_hdr
        UPDATE FIELDS ( netwr )
        WITH VALUE #( ( %tky  = ls_header_link-target
                        netwr = lv_hdr_tot ) ).
    ENDIF.

  ENDMETHOD.

  METHOD set_defaults.

    "Reader Header Entity
    READ ENTITIES OF zi_so_hdr IN LOCAL MODE
       ENTITY zi_so_hdr
       ALL FIELDS WITH CORRESPONDING #( keys )
       RESULT DATA(lt_orders).

    DATA(ls_order) = VALUE #( lt_orders[ 1 ] OPTIONAL ).

    "Read Item Entities
    READ ENTITIES OF zi_so_hdr IN LOCAL MODE
       ENTITY zi_so_item
       ALL FIELDS WITH CORRESPONDING #( keys )
       RESULT DATA(lt_items).

    "Update defaults
    MODIFY ENTITIES OF zi_so_hdr IN LOCAL MODE
      ENTITY zi_so_item
      UPDATE FIELDS ( waerk vrkme vstel )
      WITH VALUE #( FOR item IN lt_items (
                    %tky  = item-%tky
                    vrkme = 'EA'                   "Default Unit
                    waerk = 'ZAR'                  "Default Currency
                    vstel = 'Z002' ) ).            "Default Shipping Point

  ENDMETHOD.

CLASS lsc_zi_so_hdr IMPLEMENTATION.

  METHOD finalize.
  ENDMETHOD.

  METHOD check_before_save.
  ENDMETHOD.

  METHOD adjust_numbers.

    DATA: ls_header   TYPE bapisdhd1,
          lt_items    TYPE TABLE OF bapisditm,
          lt_partners TYPE TABLE OF bapiparnr,
          ls_schedule TYPE bapischdl,
          lt_schedule TYPE TABLE OF bapischdl,
          ls_cond     TYPE bapicond,
          lt_cond     TYPE TABLE OF bapicond,
          lv_vbeln    TYPE vbeln_va,
          lt_return   TYPE TABLE OF bapiret2.

    " FIX: If mapped-zi_so_hdr is empty, the framework didn't propagate entries.
    " This can happen in unmanaged scenarios. We need to build it from the buffer.
    IF mapped-zi_so_hdr IS INITIAL AND lcl_buffer=>mt_buffer IS NOT INITIAL.
      LOOP AT lcl_buffer=>mt_buffer INTO DATA(ls_buf).
        INSERT VALUE #( %pid  = ls_buf-draft_uuid
                        vbeln = ls_buf-header-vbeln ) INTO TABLE mapped-zi_so_hdr.
      ENDLOOP.
    ENDIF.

    " 1. Loop through entities waiting for a number
    LOOP AT mapped-zi_so_hdr ASSIGNING FIELD-SYMBOL(<ls_mapped_so>).

      " 2. Find the corresponding data in our global buffer
      READ TABLE lcl_buffer=>mt_buffer INTO DATA(ls_buffer_entry)
        WITH KEY draft_uuid = <ls_mapped_so>-%pid.

      if sy-subrc <> 0.
        CONTINUE.
      endif.
      " 3. Map Buffer to BAPI Header
      ls_header = VALUE #( doc_type   = ls_buffer_entry-header-auart
                           sales_org  = ls_buffer_entry-header-vkorg
                           distr_chan = ls_buffer_entry-header-vtweg
                           division   = ls_buffer_entry-header-spart
                           req_date_h = ls_buffer_entry-header-vdatu
                           price_date = ls_buffer_entry-header-crossitem_prc_date
                           doc_date   = ls_buffer_entry-header-audat
                         ).

      " 4. Map Buffer to BAPI Items
      lt_items = VALUE #( FOR item IN ls_buffer_entry-items
                          ( itm_number = item-posnr
                            material   = item-matnr
                            bill_date  = item-fkdat_ana
                            plant      = item-werks
                            target_qty = item-kwmeng
                            target_qu  = item-vrkme
                            ship_point = item-vstel
                          ) ).
      " 5. Schedule lines
      lt_schedule = VALUE #( FOR item IN ls_buffer_entry-items
                             ( itm_number = item-posnr
                               req_date   = ls_header-req_date_h
                               req_qty    = item-kwmeng
                             ) ).

      " 6. Condition Values
      lt_cond = VALUE #( FOR item IN ls_buffer_entry-items
                         ( itm_number = item-posnr
                           cond_type  = 'VBXX'
                           cond_value = item-netwr
                           currency   = 'ZAR'
                           cond_p_unt = '1'
                           curr_iso   = 'ZAR'
                           cd_unt_iso = 'EA'
                         ) ).

      " 7. Map Buffer to BAPI Partners
      lt_partners = VALUE #( FOR part IN ls_buffer_entry-partners
                             ( partn_role = part-parvw
                               partn_numb = part-kunnr
                               name       = part-name1
                               telephone  = part-telf1
                               postl_code = part-pstlz
                               country    = part-land1
                             ) ).

      " 6. Call the BAPI
      CALL FUNCTION 'BAPI_SALESORDER_CREATEFROMDAT2'
        EXPORTING
          order_header_in     = ls_header
        IMPORTING
          salesdocument       = lv_vbeln
        TABLES
          return              = lt_return
          order_items_in      = lt_items
          order_partners      = lt_partners
          order_schedules_in  = lt_schedule
          order_conditions_in = lt_cond.

      "Check if BAPI returned any Errors (Type E or A)
      IF line_exists( lt_return[ type = 'E' ] ) OR line_exists( lt_return[ type = 'A' ] ).
        "Mark the instance as FAILED to stop the save
         APPEND VALUE #( %pid = <ls_mapped_so>-%pid ) TO failed-zi_so_hdr.

          "Map BAPI messages to REPORTED
          LOOP AT lt_return INTO DATA(ls_return) WHERE type = 'E' OR type = 'A'.
            APPEND VALUE #(
                %pid = <ls_mapped_so>-%pid
                %msg = new_message( id         = ls_return-id
                                     number   = ls_return-number
                                     severity = if_abap_behv_message=>severity-error
                                     v1       = ls_return-message_v1
                                     v2       = ls_return-message_v2
                                     v3       = ls_return-message_v3
                                     v4       = ls_return-message_v4 )
            ) TO reported-zi_so_hdr.
          ENDLOOP.
      ELSE.
        <ls_mapped_so>-vbeln = lv_vbeln.
      ENDIF.
    ENDLOOP.

  ENDMETHOD.

  METHOD save.
  ENDMETHOD.

  METHOD cleanup.
      lcl_buffer=>clear( ).
  ENDMETHOD.

  METHOD cleanup_finalize.
    lcl_buffer=>clear( ).
  ENDMETHOD.
